# 4. Loop through the mtcars data frame. For each row, compute the difference
#  between the mpg column and the average. Save these differences to a vector.
#  Finally sum the values (should be equal to 0).

Use numeric() without an argument to create a new numeric vector with no elements 
mpg_avg <- mean(mtcars$mpg)
mpg_diff <- numeric()
for (i in 1:nrow(mtcars)) {
  print(i)
  mpg_diff <- c(mpg_diff, mtcars[i,"mpg"] - mpg_avg)
}
