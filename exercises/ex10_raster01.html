# 1. Import the GeoTiff file SP27GTIF.TIF that comes with the rgdal package.
#  View its properties.

To find the directory where pictures from the rgdal package are saved, run system.file("pictures", package="rgdal") 
library(raster)
img_dir <- system.file("pictures", package="rgdal")
img_fn <- file.path(img_dir, "SP27GTIF.TIF")
file.exists(img_fn)
## [1] TRUE
img_rast <- raster::raster(img_fn)
print(img_rast)

# 5. What part of the planet is this image from?

(Un)project the image to geographic coordinates (see Geographic Coordinates). Then enter the latitude and longitude into Google Maps by changing the URL http://maps.google.com/?q=36.3,-119.0 (note that Google Map URLs expect the latitude followed by the longitude). 
img_rast_ll <- projectRaster(img_rast, crs=sp::CRS("+init=epsg:4326"))
plot(img_rast_ll, col=gray.colors(255))

The image is the Chicago waterfront.
